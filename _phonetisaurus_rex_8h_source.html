45 struct VectorIntHash {
46  size_t operator () (const vector<int>& v) const {
47  size_t result = 0;
48  hash<int> hash_fn;
49 
50  for (size_t i = 0; i < v.size(); i++)
51  result ^= hash_fn (v[i]) + 0x9e3779b9 + (result << 6)
52  + (result >> 2);
53  return result;
54  }

70 int LoadClusters (const SymbolTable* syms, SymbolMap12M* clusters,
71  SymbolMapM21* invclusters) {
72  /*
73  Compute the set of 'clustered' graphemes learned during
74  the alignment process. This information is encoded in
75  the input symbols table.
76  */
77  string tie = syms->Find (1);
78  size_t max_len = 1;
79  for (size_t i = 2; i < syms->NumSymbols(); i++) {
80  string sym = syms->Find (i);
81  vector<int> cluster;
82  if (sym.find(tie) != string::npos) {
83  char* tmpstring = (char *)sym.c_str();
84  char* p = strtok (tmpstring, tie.c_str());
85  while (p) {
86  cluster.push_back (syms->Find(p));
87  p = strtok (NULL, tie.c_str());
88  }
89 
90  clusters->insert(pair<int, vector<int> >(i, cluster));
91  invclusters->insert(pair<vector<int>, int>(cluster, i));
92  max_len = (cluster.size() > max_len) ? cluster.size() : max_len;
93  } else {
94  cluster.push_back (i);
95  clusters->insert(pair<int, vector<int> >(i, cluster));
96  invclusters->insert(pair<vector<int>, int>(cluster, i));
97  }
98  }
99  return max_len;
100 }

103 void Entry2FSA (const vector<int>& word, VectorFst<Arc>* fsa, size_t maxlen,
104  const SymbolMapM21& invmap, bool superfinal=false) {
105  fsa->AddState ();
106  fsa->SetStart (0);
107  size_t j;
108  for (size_t i = 0; i < word.size(); i++) {
109  j = 1;
110  fsa->AddArc (i, Arc (word[i], word[i], Arc::Weight::One(), i + j));
111  j++;
112 
113  while (j <= maxlen && i + j <= (size_t)word.size()) {
114  vector<int> subv (&word[i], &word[i+j]);
115  SymbolMapM21::const_iterator invmap_iter = invmap.find (subv);
116  if (invmap_iter != invmap.end())
117  fsa->AddArc (i, Arc (invmap_iter->second, invmap_iter->second,
118  Arc::Weight::One(), i+j));
119  j++;
120  }
121  fsa->AddState ();
122 
123  }
124 
125  if (superfinal) {
126  fsa->AddState();
127  fsa->AddArc (word.size(), Arc (0, 0, Arc::Weight::One(), word.size()+1));
128  fsa->AddState();
129  fsa->AddArc (word.size()+1, Arc (1, 1, Arc::Weight::One(), word.size()+2));
130  fsa->SetFinal (word.size()+2, Arc::Weight::One());
131  } else {
132  fsa->SetFinal (word.size(), Arc::Weight::One());
133  }
134 }

183  void Extend (Path* path, const Arc& arc) {
184  if (arc.ilabel == 0 && arc.olabel == 0
185  && arc.weight == Arc::Weight::One())
186  return;
187 
188  SymbolMap12M::const_iterator iter = label_map_.find (arc.olabel);
189  if (iter != label_map_.end()) {
190  const vector<int>& tokens = iter->second;
191  for (int i = 0; i < tokens.size(); i++)
192  if (veto_set_.find (tokens[i]) == veto_set_.end())
193  path->unique_olabels.push_back (tokens[i]);
194  }
195 
196  path->ILabels.push_back (arc.ilabel);
197  path->OLabels.push_back (arc.olabel);
198  path->PathWeights.push_back (arc.weight.Value());
199  path->PathWeight += arc.weight.Value();
200  }

208 template<class Arc, class RevArc, class PathFilter>
209 void NShortestPathSpecialized (const Fst<RevArc> &ifst,
210  MutableFst<Arc> *ofst,
211  const vector<typename Arc::Weight> &distance,
212  size_t beam,
213  size_t nbest,
214  PathFilter* path_filter,
215  float delta = kDelta,
216  typename Arc::Weight weight_threshold = Arc::Weight::Zero (),
217  typename Arc::StateId state_threshold = kNoStateId) {
218  typedef typename Arc::StateId StateId;
219  typedef typename Arc::Weight Weight;
220  typedef pair<StateId, Weight> Pair;
221  //typedef typename RevArc::Weight RevWeight;
222 
223  if (nbest <= 0) return;
224  if ((Weight::Properties () & (kPath | kSemiring)) != (kPath | kSemiring)) {
225  FSTERROR() << "NShortestPath: Weight needs to have the "
226  << "path property and be distributive: "
227  << Weight::Type ();
228  ofst->SetProperties (kError, kError);
229  return;
230  }
231  ofst->DeleteStates ();
232  ofst->SetInputSymbols (ifst.InputSymbols ());
233  ofst->SetOutputSymbols (ifst.OutputSymbols ());
234  // Each state in 'ofst' corresponds to a path with weight w from the
235  // initial state of 'ifst' to a state s in 'ifst', that can be
236  // characterized by a pair (s,w). The vector 'pairs' maps each
237  // state in 'ofst' to the corresponding pair maps states in OFST to
238  // the corresponding pair (s,w).
239  vector<Pair> pairs;
240  // The supefinal state is denoted by -1, 'compare' knows that the
241  // distance from 'superfinal' to the final state is 'Weight::One()',
242  // hence 'distance[superfinal]' is not needed.
243  StateId superfinal = -1;
244  ShortestPathCompare<StateId, Weight>
245  compare(pairs, distance, superfinal, delta);
246  vector<StateId> heap;
247  // 'r[s + 1]', 's' state in 'fst', is the number of states in 'ofst'
248  // which corresponding pair contains 's' ,i.e. , it is number of
249  // paths computed so far to 's'. Valid for 's == -1' (superfinal).
250  vector<int> r;
251  NaturalLess<Weight> less;
252  if (ifst.Start () == kNoStateId ||
253  distance.size () <= ifst.Start () ||
254  distance [ifst.Start ()] == Weight::Zero () ||
255  less (weight_threshold, Weight::One ()) ||
256  state_threshold == 0) {
257  if (ifst.Properties (kError, false)) ofst->SetProperties (kError, kError);
258  return;
259  }
260  ofst->SetStart (ofst->AddState ());
261  StateId final = ofst->AddState ();
262  ofst->SetFinal(final, Weight::One ());
263  while (pairs.size () <= final)
264  pairs.push_back (Pair (kNoStateId, Weight::Zero ()));
265  pairs [final] = Pair (ifst.Start(), Weight::One ());
266  heap.push_back (final);
267  Weight limit = Times (distance [ifst.Start ()], weight_threshold);
268 
269  // Treat 'n' like a beam. npaths contains target number of unique paths
270  while (!heap.empty ()) {
271  pop_heap (heap.begin (), heap.end (), compare);
272  StateId state = heap.back ();
273  Pair p = pairs [state];
274  heap.pop_back ();
275  Weight d = p.first == superfinal ? Weight::One () :
276  p.first < distance.size () ? distance [p.first] : Weight::Zero ();
277 
278  if (less (limit, Times (d, p.second)) ||
279  (state_threshold != kNoStateId &&
280  ofst->NumStates () >= state_threshold))
281  continue;
282 
283  while (r.size () <= p.first + 1) r.push_back (0);
284  ++r[p.first + 1];
285 
286  // This is extended to 'filter' for unique paths based
287  // on some sort of user-defined function. Unlike the
288  // existing 'unique' functionality in the original this
289  // provides for an arbitrary definition of uniqueness, and
290  // by avoiding determinization ensures that ngram-weights
291  // will not be moved or smeared
292  if (p.first == superfinal) {
293  ofst->AddArc(ofst->Start (), Arc (0, 0, Weight::One (), state));
294  StateId tstate = state;
295  Path one_path;
296  while (ofst->Final (tstate) == Weight::Zero ()) {
297  for (ArcIterator<Fst<Arc> > aiter (*ofst, tstate);
298  !aiter.Done (); aiter.Next ()) {
299  const Arc& tarc = aiter.Value ();
300  tstate = tarc.nextstate;
301  path_filter->Extend (&one_path, tarc);
302  }
303  }
304 
305  if (path_filter->path_map.find (one_path.unique_olabels)
306  == path_filter->path_map.end ()) {
307  path_filter->path_map.insert(
308  pair<vector<int>, Path> (one_path.unique_olabels, one_path));
309  path_filter->ordered_paths.push_back (one_path.unique_olabels);
310 
311  if (path_filter->ordered_paths.size () >= nbest)
312  break;
313  }
314  }
315 
316  if ((p.first == superfinal) && (r [p.first + 1] == beam))
317  break;
318 
319  if (r [p.first + 1] > beam) continue;
320  if (p.first == superfinal) continue;
321  for (ArcIterator< Fst<RevArc> > aiter (ifst, p.first);
322  !aiter.Done ();
323  aiter.Next ()) {
324  const RevArc &rarc = aiter.Value();
325  Arc arc(rarc.ilabel, rarc.olabel, rarc.weight.Reverse (), rarc.nextstate);
326  Weight w = Times (p.second, arc.weight);
327  StateId next = ofst->AddState ();
328  pairs.push_back (Pair (arc.nextstate, w));
329  arc.nextstate = state;
330  ofst->AddArc (next, arc);
331  heap.push_back (next);
332  push_heap (heap.begin (), heap.end(), compare);
333  }
334 
335  Weight finalw = ifst.Final (p.first).Reverse ();
336  if (finalw != Weight::Zero ()) {
337  Weight w = Times (p.second, finalw);
338  StateId next = ofst->AddState ();
339  pairs.push_back (Pair (superfinal, w));
340  ofst->AddArc (next, Arc(0, 0, finalw, state));
341  heap.push_back (next);
342  push_heap (heap.begin (), heap.end (), compare);
343  }
344  }
345  Connect (ofst);
346  if (ifst.Properties (kError, false)) ofst->SetProperties (kError, kError);
347  ofst->SetProperties (
348  ShortestPathProperties (ofst->Properties (kFstProperties, false)),
349  kFstProperties);
350 }

353 template<class Arc, class Queue, class ArcFilter, class PathFilter>
354 void ShortestPathSpecialized(const Fst<Arc> &ifst, MutableFst<Arc> *ofst,
355  vector<typename Arc::Weight> *distance,
356  PathFilter* path_filter,
357  size_t beam,
358  ShortestPathOptions<Arc, Queue, ArcFilter> &opts) {
359  typedef typename Arc::StateId StateId;
360  typedef typename Arc::Weight Weight;
361  typedef ReverseArc<Arc> ReverseArc;
362 
363  size_t nbest = opts.nshortest;
364  if (nbest <= 0) return;
365  if ((Weight::Properties () & (kPath | kSemiring)) != (kPath | kSemiring)) {
366  FSTERROR() << "ShortestPath: n-shortest: Weight needs to have the "
367  << "path property and be distributive: "
368  << Weight::Type ();
369  ofst->SetProperties (kError, kError);
370  return;
371  }
372  if (!opts.has_distance) {
373  ShortestDistance (ifst, distance, opts);
374  if (distance->size () == 1 && !(*distance) [0].Member ()) {
375  ofst->SetProperties (kError, kError);
376  return;
377  }
378  }
379  // Algorithm works on the reverse of 'fst' : 'rfst', 'distance' is
380  // the distance to the final state in 'rfst', 'ofst' is built as the
381  // reverse of the tree of n-shortest path in 'rfst'.
382  VectorFst<ReverseArc> rfst;
383  Reverse(ifst, &rfst);
384  Weight d = Weight::Zero();
385  for (ArcIterator< VectorFst<ReverseArc> > aiter(rfst, 0);
386  !aiter.Done (); aiter.Next ()) {
387  const ReverseArc &arc = aiter.Value ();
388  StateId s = arc.nextstate - 1;
389  if (s < distance->size())
390  d = Plus (d, Times (arc.weight.Reverse(), (*distance)[s]));
391  }
392  distance->insert (distance->begin(), d);
393 
394  //Specialize the 'uniqueness' property for our needs
395  NShortestPathSpecialized (rfst, ofst, *distance, beam, nbest, path_filter, opts.delta,
396  opts.weight_threshold, opts.state_threshold);
397 
398  distance->erase (distance->begin());
399 }
