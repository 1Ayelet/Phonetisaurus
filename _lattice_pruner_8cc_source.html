55 void LatticePruner::prune_fst( VectorFst<StdArc>* fst ){
56  /*
57  Apply several optional pruning heuristics to the lattice.
58  */
59  if( penalize==true )
60  _penalize_arcs( fst );
61 
62  if( fb==true )
63  _forward_backward( fst );
64 
65  if( nbest==1 ){
66  //If N=1 then all the remaining stuff is a waste of time.
67  //This is because the pruning heuristics are all computed
68  // *relative* to the 1-best hypothesis.
69  //This is in contrast LMBR and arc penalization.
70  *fst = _nbest_prune( fst );
71  return;
72  }
73 
74 
75  if( beam.Value() != LogWeight::Zero() )
76  Prune( fst, beam );
77 
78  if( nbest > 1 )
79  *fst = _nbest_prune( fst );
80 
81  return;
82 }

97 void LatticePruner::_forward_backward( VectorFst<StdArc>* fst ){
98  /*
99  OpenFst-based implementation of forward-backward lattice pruning based on,
100  Sixtus and Ortmanns, "HIGH QUALITY WORD GRAPHS USING FORWARD-BACKWARD PRUNING", 1999
101 
102  Note-to-self: It seems to give consistent WER and PER improvements so I guess I
103  got the implementation right, but it seems like maybe it was too easy.
104  */
105  //Setup
106  VectorFst<LogArc>* pfst = new VectorFst<LogArc>();
107  VectorFst<LogArc>* lfst = new VectorFst<LogArc>();
108  vector<LogWeight> alpha, beta;
109 
110  Map(*fst, lfst, StdToLogMapper());
111 
112  //Normalize so that subsequent operations don't go crazy
113  Push<LogArc, REWEIGHT_TO_FINAL>(*lfst, pfst, kPushWeights);
114  for( StateIterator<VectorFst<LogArc> > siter(*pfst); !siter.Done(); siter.Next() ){
115  size_t i = siter.Value();
116  if( pfst->Final(i)!=LogArc::Weight::Zero() ){
117  pfst->SetFinal(i,LogArc::Weight::One());
118  }
119  }
120 
121  //Compute Forward and Backward probabilities
122  ShortestDistance( *pfst, &alpha );
123  ShortestDistance( *pfst, &beta, true );
124 
125  //Compute arc posteriors. This is the same as the Expectation step.
126  for( StateIterator<VectorFst<LogArc> > siter(*pfst); !siter.Done(); siter.Next() ){
127  LogArc::StateId q = siter.Value();
128  for( MutableArcIterator<VectorFst<LogArc> > aiter(pfst,q); !aiter.Done(); aiter.Next() ){
129  LogArc arc = aiter.Value();
130  LogWeight gamma = Divide(Times(Times(alpha[q], arc.weight), beta[arc.nextstate]), beta[0]);
131 
132  if( gamma.Value()==gamma.Value() ){
133  arc.weight = gamma;
134  aiter.SetValue(arc);
135  }
136  }
137  }
138 
139  Map(*pfst, fst, LogToStdMapper());
140 
141  delete lfst;
142  delete pfst;
143  return;
144 }

146 void LatticePruner::_penalize_arcs( VectorFst<StdArc>* fst ){
147 
148  for( StateIterator<VectorFst<StdArc> > siter(*fst); !siter.Done(); siter.Next() ){
149  StdArc::StateId q = siter.Value();
150  for( MutableArcIterator<VectorFst<StdArc> > aiter(fst,q); !aiter.Done(); aiter.Next() ){
151  StdArc arc = aiter.Value();
152  LabelDatum* ld = &penalties[arc.ilabel];
153 
154  if( ld->lhs>1 && ld->rhs>1 ){
155  arc.weight = 999;
156  }else{
157  arc.weight = arc.weight.Value() * ld->max;
158  }
159  if( arc.weight == LogWeight::Zero() )
160  arc.weight = 999;
161  if( arc.weight != arc.weight )
162  arc.weight = 999;
163  aiter.SetValue(arc);
164  }
165  }
166 
167  return;
168 }
