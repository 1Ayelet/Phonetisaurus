106  // The actual phoneticizer routine
107  vector<PathData> Phoneticize (const string& word, int nbest = 1,
108  int beam = 10000, float threshold = 99,
109  bool write_fsts = false) {
110  VectorFst<StdArc>* fst = new VectorFst<StdArc> ();
111  vector<int> entry = tokenize2ints ((string*) &word, &delim_, isyms_);
112  Entry2FSA (entry, fst, imax_, invimap_);
113 
114  fst->SetInputSymbols (isyms_);
115  fst->SetOutputSymbols (isyms_);
116 
117  //Useful for debugging
118  if (write_fsts)
119  fst->Write (word+".fst");
120 
121  VectorFst<StdArc> ofst;
122 
123  StdArc::Weight weight_threshold = threshold;
124  StdArc::StateId state_threshold = kNoStateId;
125  AnyArcFilter<StdArc> arc_filter;
126  vector<StdArc::Weight> distance;
127 
128  //ComposeFst<StdArc>* ifst = new ComposeFst<StdArc>(*fst, model_);
129  VectorFst<StdArc>* ifst = new VectorFst<StdArc>();
130  Compose(*fst, model_, ifst);
131  //Useful for debugging
132  if (write_fsts)
133  ifst->Write (word+".lat.fst");
134 
135  AutoQueue<StdArc::StateId> state_queue (*ifst, &distance, arc_filter);
136 
137  M2MPathFilter<StdArc> path_filter (omap_, veto_set_);
138 
139  ShortestPathOptions<StdArc, AutoQueue<StdArc::StateId>,
140  AnyArcFilter<StdArc> >
141  opts (&state_queue, arc_filter, nbest, false, false,
142  kDelta, false, weight_threshold,
143  state_threshold);
144 
145  ShortestPathSpecialized (*ifst, &ofst, &distance,
146  &path_filter, beam, opts);
147 
148  vector<PathData> paths;
149  for (size_t i = 0; i < path_filter.ordered_paths.size(); i++) {
150  const vector<int>& u = path_filter.ordered_paths[i];
151  const Path& orig = path_filter.path_map[u];
152  PathData path = PathData (orig.PathWeight, orig.PathWeights,
153  orig.ILabels, orig.OLabels, orig.unique_olabels);
154  paths.push_back (path);
155  }
156 
157  // Make sure that we clean up
158  delete fst;
159  delete ifst;
160  return paths;
161  }
